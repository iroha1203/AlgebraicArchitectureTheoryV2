  rw [pow_two]
  apply le_antisymm
  · intro z hz
    change z = 0
    refine Submodule.mul_induction_on hz ?_ ?_
    · intro x hx y hy
      change TrivSqZeroExt.fst x = 0 at hx
      change TrivSqZeroExt.fst y = 0 at hy
      apply TrivSqZeroExt.ext
      · simp [hx, hy]
      · simp [hx, hy]
    · intro x y hx hy
      rw [hx, hy, add_zero]
  · exact bot_le

def inrToTrivialExtensionIdeal : B →ₗ[B] trivialExtensionIdeal B :=
  { toFun := fun b ↦ ⟨TrivSqZeroExt.inr b, by
      change TrivSqZeroExt.fst (TrivSqZeroExt.inr b) = 0
      simp⟩
    map_add' := fun x y ↦ by
      apply Subtype.ext
      exact TrivSqZeroExt.inr_add B x y
    map_smul' := fun r x ↦ by
      apply Subtype.ext
      exact TrivSqZeroExt.inr_smul B r x }

    Function.Bijective (inrToTrivialExtensionIdeal B) := by

    have hsnd := congrArg
      (fun z : TrivSqZeroExt B B ↦ TrivSqZeroExt.snd z)
      (congrArg Subtype.val h)

## Cycle 12 — L0 finite linear repair and support-minimal circuit

- decision: approve
- result type: proof-obligation-discharged
- proof obligation: prove the fiberwise finite linear repair alternative,
  extract a target-containing support-minimal dependence on failure, and
  separate the loop and nonloop branches
- base merge SHA: `2b25dbae7408bf2eb4284226c77ccf740cc1ef16`
- PR: #3324

### Construction

`LinearRepair.lean` fixes a finite protected family of response functionals
`response p : E →ₗ[K] K`. A normalized repair is an actual vector `repair : E`
that annihilates every protected response and evaluates the target response to
`1`. The finite common-kernel theorem constructs such a vector exactly when
the target response functional is outside the protected response span. Its
negation gives target span membership without accepting a repair or image
membership certificate as input.

For an arbitrary finite protected label set, target span membership is
equivalent to the existence of a finite inclusion-minimal linearly dependent
set `C` with

```text
target ∈ C ⊆ P ∪ {target}.
```

The proof first extracts an independent subset of the protected labels that
spans all protected responses, then minimizes among the dependent subsets of
that independent set with the target inserted. This prevents an unrelated
dependence already present inside `P` from removing the target from the
resulting circuit.

If the target response is zero, the singleton target set is the loop circuit
and normalized repair is impossible. If the target response is nonzero and
repair fails, the extracted circuit contains at least one protected label.

  - normalized repair existence iff target response is outside the protected
